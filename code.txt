class PairedImageDataset(Dataset):
    def __init__(self, root_dir, transform=None):
        self.transform = transform
        pre_dir = os.path.join(root_dir, "pre")
        post_dir = os.path.join(root_dir, "post")
        pre_op_files = sorted(glob.glob(os.path.join(pre_dir, "*.png")))
        
        self.pre_op_files = []
        self.post_op_files = []

        for pre_path in pre_op_files:
            filename = os.path.basename(pre_path)
            if '_pre_' in filename:
                post_filename = filename.replace('_pre_', '_post_')
            elif '_pre.' in filename:
                post_filename = filename.replace('_pre.', '_post.')
            else:
                post_filename = filename
            
            post_path = os.path.join(post_dir, post_filename)
            if os.path.exists(post_path):
                self.pre_op_files.append(pre_path)
                self.post_op_files.append(post_path)

    def __len__(self):
        return len(self.pre_op_files)

    def __getitem__(self, idx):
        pre_op_img = Image.open(self.pre_op_files[idx]).convert("L")
        post_op_img = Image.open(self.post_op_files[idx]).convert("L")

        if self.transform:
            seed = np.random.randint(2147483647)
            torch.manual_seed(seed)
            pre_op_img = self.transform(pre_op_img)
            torch.manual_seed(seed)
            post_op_img = self.transform(post_op_img)

        return pre_op_img, post_op_img

# ---- Replace existing SelfAttention with this T4-friendly pooled attention ----
class SelfAttention(nn.Module):
    def __init__(self, channels, pool_size=16, proj_dim=None):
        """
        pooled attention: compute K,V on a pooled spatial grid (pool_size),
        keep Q at full res so locality is preserved when adding back.
        pool_size: target spatial size for k/v (e.g. 16)
        proj_dim: internal projection dimension per head (optional)
        """
        super().__init__()
        self.channels = channels
        self.pool_size = pool_size

        # projection dims
        if proj_dim is None:
            proj_dim = channels  # no channel reduction by default

        self.to_q = nn.Conv2d(channels, proj_dim, 1, bias=False)
        self.to_k = nn.Conv2d(channels, proj_dim, 1, bias=False)
        self.to_v = nn.Conv2d(channels, proj_dim, 1, bias=False)
        self.to_out = nn.Conv2d(proj_dim, channels, 1)

        self.ln = nn.LayerNorm(proj_dim)

    def forward(self, x):
        # x: (B, C, H, W)
        B, C, H, W = x.shape

        # Q at full resolution
        q = self.to_q(x)                       # (B, proj_dim, H, W)
        q_flat = q.view(B, -1, H*W).transpose(1, 2)  # (B, N, Dq)

        # Pool spatially for K/V to reduce complexity
        # Use adaptive_avg_pool2d to target approximately pool_size x pool_size
        pool_h = min(self.pool_size, H)
        pool_w = min(self.pool_size, W)
        kv = F.adaptive_avg_pool2d(x, (pool_h, pool_w))  # (B, C, pool_h, pool_w)

        k = self.to_k(kv)                       # (B, Dk, pool_h, pool_w)
        v = self.to_v(kv)                       # (B, Dv, pool_h, pool_w)

        k_flat = k.view(B, -1, pool_h*pool_w).transpose(1, 2)  # (B, M, Dk)
        v_flat = v.view(B, -1, pool_h*pool_w).transpose(1, 2)  # (B, M, Dv)

        # LayerNorm on feature dim (apply to q/k/v; LN expects last dim)
        q_ln = self.ln(q_flat)
        k_ln = self.ln(k_flat)

        # Scaled dot-product attention (small M)
        # q_ln: (B, N, D), k_ln: (B, M, D), v_flat: (B, M, D)
        attn_scores = torch.matmul(q_ln, k_ln.transpose(1, 2))  # (B, N, M)
        attn_scores = attn_scores / math.sqrt(q_ln.shape[-1])
        attn_weights = torch.softmax(attn_scores, dim=-1)      # (B, N, M)

        out = torch.matmul(attn_weights, v_flat)               # (B, N, D)
        out = out.transpose(1, 2).view(B, -1, H, W)            # (B, D, H, W)

        out = self.to_out(out)                                 # (B, C, H, W)
        return x + out


# ===== TIME EMBEDDING =====
class TimeEmbedding(nn.Module):
    def __init__(self, dim):
        super().__init__()
        self.dim = dim
        self.mlp = nn.Sequential(
            nn.Linear(dim, dim * 4),
            nn.GELU(),
            nn.Linear(dim * 4, dim)
        )
    
    def forward(self, t):
        half_dim = self.dim // 2
        embeddings = math.log(10000) / (half_dim - 1)
        embeddings = torch.exp(torch.arange(half_dim, device=t.device) * -embeddings)
        embeddings = t[:, None] * embeddings[None, :]
        embeddings = torch.cat([embeddings.sin(), embeddings.cos()], dim=-1)
        return self.mlp(embeddings)

# ===== FIXED RESIDUAL BLOCK =====
class ResidualBlock(nn.Module):
    def __init__(self, in_channels, out_channels, time_emb_dim, dropout=0.1, num_groups=4):
        super().__init__()
        self.time_mlp = nn.Linear(time_emb_dim, out_channels)
        
        # FIX: Ensure in_channels is divisible by num_groups
        self.adjusted_in_channels = self._adjust_channels(in_channels, num_groups)
        
        # Add projection layer if needed
        if in_channels != self.adjusted_in_channels:
            self.channel_adjust = nn.Conv2d(in_channels, self.adjusted_in_channels, 1)
        else:
            self.channel_adjust = nn.Identity()
        
        self.block1 = nn.Sequential(
            nn.GroupNorm(num_groups, self.adjusted_in_channels),
            nn.SiLU(),
            nn.Conv2d(self.adjusted_in_channels, out_channels, 3, padding=1)
        )
        
        self.block2 = nn.Sequential(
            nn.GroupNorm(num_groups, out_channels),
            nn.SiLU(),
            nn.Dropout(dropout),
            nn.Conv2d(out_channels, out_channels, 3, padding=1)
        )
        
        if self.adjusted_in_channels != out_channels:
            self.shortcut = nn.Conv2d(self.adjusted_in_channels, out_channels, 1)
        else:
            self.shortcut = nn.Identity()
    
    def _adjust_channels(self, channels, num_groups):
        """Adjust channels to be divisible by num_groups"""
        if channels % num_groups == 0:
            return channels
        return ((channels + num_groups - 1) // num_groups) * num_groups
    
    def forward(self, x, t):
        x = self.channel_adjust(x)
        h = self.block1(x)
        time_emb = self.time_mlp(t)
        h = h + time_emb[:, :, None, None]
        h = self.block2(h)
        return h + self.shortcut(x)

# ===== CONDITIONAL U-NET =====
class ConditionalUNet(nn.Module):
    def __init__(self, in_channels=1, out_channels=1, time_emb_dim=256, 
                 base_channels=64, channel_mults=(1, 2, 4, 8), num_groups=4):
        super().__init__()
        
        self.time_mlp = TimeEmbedding(time_emb_dim)
        
        self.condition_encoder = nn.Sequential(
            nn.Conv2d(in_channels, base_channels, 3, padding=1),
            nn.GroupNorm(num_groups, base_channels),
            nn.SiLU()
        )
        
        self.input_proj = nn.Conv2d(in_channels * 2, base_channels, 3, padding=1)
        
        self.encoder_blocks = nn.ModuleList()
        self.encoder_downs = nn.ModuleList()
        
        now_channels = base_channels
        skip_channels = []
        
        for i, mult in enumerate(channel_mults):
            out_channels_block = base_channels * mult
            level_blocks = nn.ModuleList()
            for _ in range(2):
                level_blocks.append(ResidualBlock(
                    now_channels + base_channels,
                    out_channels_block,
                    time_emb_dim,
                    num_groups=num_groups
                ))
                skip_channels.append(out_channels_block)
                now_channels = out_channels_block
            
            self.encoder_blocks.append(level_blocks)
            
            if i != len(channel_mults) - 1:
                self.encoder_downs.append(nn.Conv2d(now_channels, now_channels, 3, 2, 1))
                skip_channels.append(now_channels)
            else:
                self.encoder_downs.append(None)
        
        self.mid = nn.ModuleList([
            ResidualBlock(now_channels, now_channels, time_emb_dim, num_groups=num_groups),
            SelfAttention(now_channels, pool_size=16, proj_dim=now_channels//1),
            ResidualBlock(now_channels, now_channels, time_emb_dim, num_groups=num_groups)
        ])
        
        self.decoder_blocks = nn.ModuleList()
        self.decoder_ups = nn.ModuleList()
        
        skip_channels = list(reversed(skip_channels))
        skip_idx = 0
        
        for i, mult in enumerate(reversed(channel_mults)):
            out_channels_block = base_channels * mult
            level_blocks = nn.ModuleList()
            for j in range(3):
                if skip_idx < len(skip_channels):
                    in_ch = skip_channels[skip_idx] + now_channels
                    skip_idx += 1
                else:
                    in_ch = now_channels
                
                level_blocks.append(ResidualBlock(
                    in_ch,
                    out_channels_block,
                    time_emb_dim,
                    num_groups=num_groups
                ))
                now_channels = out_channels_block
            
            self.decoder_blocks.append(level_blocks)
            
            if i != len(channel_mults) - 1:
                self.decoder_ups.append(nn.ConvTranspose2d(now_channels, now_channels, 4, 2, 1))
            else:
                self.decoder_ups.append(None)
        
        self.out = nn.Sequential(
            nn.GroupNorm(num_groups, now_channels),
            nn.SiLU(),
            nn.Conv2d(now_channels, 1, 3, padding=1)
        )
    
    def forward(self, x, condition, t, x_self_cond=None):
        t_emb = self.time_mlp(t)
        condition_features = self.condition_encoder(condition)
        
        if x_self_cond is not None:
            x = torch.cat([x, x_self_cond], dim=1)
        else:
            x = torch.cat([x, torch.zeros_like(x)], dim=1)
        
        h = self.input_proj(x)
        
        skips = []
        for blocks, downsample in zip(self.encoder_blocks, self.encoder_downs):
            for block in blocks:
                h = torch.cat([h, condition_features], dim=1)
                h = block(h, t_emb)
                skips.append(h)
            
            if downsample is not None:
                h = downsample(h)
                condition_features = F.interpolate(condition_features, scale_factor=0.5, mode='bilinear')
                skips.append(h)
        
        for module in self.mid:
            if isinstance(module, ResidualBlock):
                h = module(h, t_emb)
            else:
                h = module(h)
        
        for blocks, upsample in zip(self.decoder_blocks, self.decoder_ups):
            for block in blocks:
                if skips:
                    h = torch.cat([h, skips.pop()], dim=1)
                h = block(h, t_emb)
            
            if upsample is not None:
                h = upsample(h)
        
        return self.out(h)

# ===== DIFFUSION PROCESS =====
class DiffusionModel:
    def __init__(self, timesteps=1000, beta_start=0.0001, beta_end=0.02, device='cuda'):
        self.timesteps = timesteps
        self.device = device
        
        self.betas = torch.linspace(beta_start, beta_end, timesteps).to(device)
        self.alphas = 1 - self.betas
        self.alphas_cumprod = torch.cumprod(self.alphas, dim=0)
        self.alphas_cumprod_prev = F.pad(self.alphas_cumprod[:-1], (1, 0), value=1.0)
        
        self.sqrt_alphas_cumprod = torch.sqrt(self.alphas_cumprod)
        self.sqrt_one_minus_alphas_cumprod = torch.sqrt(1 - self.alphas_cumprod)
        self.sqrt_recip_alphas = torch.sqrt(1 / self.alphas)
        self.posterior_variance = self.betas * (1 - self.alphas_cumprod_prev) / (1 - self.alphas_cumprod)
    
    def q_sample(self, x_start, t, noise=None):
        if noise is None:
            noise = torch.randn_like(x_start)
        
        sqrt_alphas_cumprod_t = self.sqrt_alphas_cumprod[t][:, None, None, None]
        sqrt_one_minus_alphas_cumprod_t = self.sqrt_one_minus_alphas_cumprod[t][:, None, None, None]
        
        return sqrt_alphas_cumprod_t * x_start + sqrt_one_minus_alphas_cumprod_t * noise
    
    def predict_start_from_noise(self, x_t, t, noise):
        sqrt_alphas_cumprod_t = self.sqrt_alphas_cumprod[t][:, None, None, None]
        sqrt_one_minus_alphas_cumprod_t = self.sqrt_one_minus_alphas_cumprod[t][:, None, None, None]
        
        return (x_t - sqrt_one_minus_alphas_cumprod_t * noise) / sqrt_alphas_cumprod_t
    
    @torch.no_grad()
    def ddim_sample(self, model, condition, steps=50, eta=0.0):
        model.eval()
        b = condition.shape[0]
        device = condition.device
        
        skip = self.timesteps // steps
        seq = range(0, self.timesteps, skip)
        seq_next = [-1] + list(seq[:-1])
        
        x = torch.randn(b, 1, config.IMG_SIZE, config.IMG_SIZE, device=device)
        x_self_cond = None
        
        for i, j in zip(reversed(seq), reversed(seq_next)):
            t = torch.full((b,), i, device=device, dtype=torch.long)
            
            predicted_noise = model(x, condition, t, x_self_cond)
            
            if config.SELF_CONDITION and i % 2 == 0:
                x_self_cond = self.predict_start_from_noise(x, t, predicted_noise)
            else:
                x_self_cond = None
            
            alpha = self.alphas_cumprod[i]
            alpha_next = self.alphas_cumprod[j] if j >= 0 else torch.tensor(1.0)
            
            sigma = eta * torch.sqrt((1 - alpha_next) / (1 - alpha) * (1 - alpha / alpha_next))
            
            pred_x0 = self.predict_start_from_noise(x, t, predicted_noise)
            pred_x0 = torch.clamp(pred_x0, -1, 1)
            
            dir_xt = torch.sqrt(1 - alpha_next - sigma**2) * predicted_noise
            
            if j >= 0:
                noise = torch.randn_like(x)
                x = torch.sqrt(alpha_next) * pred_x0 + dir_xt + sigma * noise
            else:
                x = torch.sqrt(alpha_next) * pred_x0 + dir_xt
        
        return x

# ===== EMA =====
class EMA:
    def __init__(self, model, decay=0.9999):
        self.model = model
        self.decay = decay
        self.shadow = {}
        self.backup = {}
        
        for name, param in model.named_parameters():
            if param.requires_grad:
                self.shadow[name] = param.data.clone()
    
    def update(self):
        for name, param in self.model.named_parameters():
            if param.requires_grad:
                self.shadow[name] = self.decay * self.shadow[name] + (1 - self.decay) * param.data
    
    def apply_shadow(self):
        for name, param in self.model.named_parameters():
            if param.requires_grad:
                self.backup[name] = param.data.clone()
                param.data = self.shadow[name]
    
    def restore(self):
        for name, param in self.model.named_parameters():
            if param.requires_grad:
                param.data = self.backup[name]
        self.backup = {}

# ===== TRAINING =====
def train_fn(model, diffusion, loader, optimizer, scaler, epoch, ema=None):
    model.train()
    loop = tqdm(loader, leave=True, desc=f"Epoch {epoch}")
    total_loss = 0
    
    for idx, (condition, target) in enumerate(loop):
        condition = condition.to(config.DEVICE, memory_format=torch.channels_last, non_blocking=True)
        target = target.to(config.DEVICE, memory_format=torch.channels_last, non_blocking=True)

        
        t = torch.randint(0, config.TIMESTEPS, (condition.shape[0],), device=config.DEVICE).long()
        
        noise = torch.randn_like(target)
        x_noisy = diffusion.q_sample(target, t, noise)
        
        x_self_cond = None
        if config.SELF_CONDITION and np.random.random() < 0.5:
            with torch.no_grad():
                x_self_cond = model(x_noisy, condition, t, None)
                x_self_cond = diffusion.predict_start_from_noise(x_noisy, t, x_self_cond).detach()
        
        with autocast(device_type='cuda', enabled=config.MIXED_PRECISION):
            predicted_noise = model(x_noisy, condition, t, x_self_cond)
            loss = F.mse_loss(predicted_noise, noise)
        
        optimizer.zero_grad()
        scaler.scale(loss).backward()
        scaler.unscale_(optimizer)
        torch.nn.utils.clip_grad_norm_(model.parameters(), config.GRADIENT_CLIP)
        scaler.step(optimizer)
        scaler.update()
        
        if ema is not None:
            ema.update()
        
        total_loss += loss.item()
        
        if idx % 20 == 0:
            loop.set_postfix(loss=f"{loss.item():.4f}")
    
    return total_loss / len(loader)

@torch.no_grad()
def validate_with_metrics(model, diffusion, val_loader, epoch):
    model.eval()
    
    all_metrics = {'mae': [], 'mse': [], 'rmse': [], 'psnr': [], 'ssim': []}
    num_samples = min(10, len(val_loader))
    
    for idx, (condition, target) in enumerate(tqdm(val_loader, desc="Validating", total=num_samples)):
        if idx >= num_samples:
            break
        
        condition = condition.to(config.DEVICE, non_blocking=True)
        target = target.to(config.DEVICE, non_blocking=True)

        
        generated = diffusion.ddim_sample(model, condition, steps=config.SAMPLING_STEPS)
        
        pred_01 = (generated.cpu().numpy() + 1) / 2
        target_01 = (target.cpu().numpy() + 1) / 2
        
        pred_255 = (pred_01 * 255).clip(0, 255).astype(np.uint8).squeeze()
        target_255 = (target_01 * 255).clip(0, 255).astype(np.uint8).squeeze()
        
        mae = np.abs(pred_01 - target_01).mean()
        mse = ((pred_01 - target_01) ** 2).mean()
        rmse = np.sqrt(mse)
        psnr_score = psnr(target_255, pred_255, data_range=255)
        ssim_score = ssim(target_255, pred_255, data_range=255)
        
        all_metrics['mae'].append(mae)
        all_metrics['mse'].append(mse)
        all_metrics['rmse'].append(rmse)
        all_metrics['psnr'].append(psnr_score)
        all_metrics['ssim'].append(ssim_score)
        
        if idx < 3:
            fig, axes = plt.subplots(1, 4, figsize=(16, 4))
            
            cond_vis = (condition.cpu().squeeze().numpy() + 1) / 2
            target_vis = target_01.squeeze()
            gen_vis = pred_01.squeeze()
            diff = np.abs(target_vis - gen_vis)
            
            axes[0].imshow(cond_vis, cmap='gray')
            axes[0].set_title('Pre-Op (Input)')
            axes[0].axis('off')
            
            axes[1].imshow(target_vis, cmap='gray')
            axes[1].set_title('Ground Truth')
            axes[1].axis('off')
            
            axes[2].imshow(gen_vis, cmap='gray')
            axes[2].set_title('Generated')
            axes[2].axis('off')
            
            axes[3].imshow(diff, cmap='hot')
            axes[3].set_title('Difference')
            axes[3].axis('off')
            
            fig.suptitle(f'Epoch {epoch} - Sample {idx+1}')
            plt.tight_layout()
            plt.savefig(os.path.join(config.VALIDATION_DIR, f"epoch_{epoch}_sample_{idx}.png"), dpi=150)
            plt.close()
    
    avg_mae = np.mean(all_metrics['mae'])
    avg_mse = np.mean(all_metrics['mse'])
    avg_rmse = np.mean(all_metrics['rmse'])
    avg_psnr = np.mean(all_metrics['psnr'])
    avg_ssim = np.mean(all_metrics['ssim'])
    
    print(f"\n📊 Validation Results (Average of {num_samples} samples):")
    print(f"{'─'*70}")
    print(f"  MAE:  {avg_mae:.6f} ± {np.std(all_metrics['mae']):.6f}")
    print(f"  MSE:  {avg_mse:.6f} ± {np.std(all_metrics['mse']):.6f}")

@torch.no_grad()
def run_final_testing(model, diffusion, test_loader):
    print("\n" + "="*70)
    print("FINAL TESTING")
    print("="*70)
    
    model.eval()
    all_metrics = {'mae': [], 'mse': [], 'rmse': [], 'psnr': [], 'ssim': []}
    
    for idx, (condition, target) in enumerate(tqdm(test_loader, desc="Testing")):
        condition = condition.to(config.DEVICE, non_blocking=True)
        target = target.to(config.DEVICE, non_blocking=True)

        
        generated = diffusion.ddim_sample(model, condition, steps=config.SAMPLING_STEPS)
        
        pred_01 = (generated.cpu().numpy() + 1) / 2
        target_01 = (target.cpu().numpy() + 1) / 2
        
        pred_255 = (pred_01 * 255).clip(0, 255).astype(np.uint8).squeeze()
        target_255 = (target_01 * 255).clip(0, 255).astype(np.uint8).squeeze()
        
        mae = np.abs(pred_01 - target_01).mean()
        mse = ((pred_01 - target_01) ** 2).mean()
        rmse = np.sqrt(mse)
        psnr_score = psnr(target_255, pred_255, data_range=255)
        ssim_score = ssim(target_255, pred_255, data_range=255)
        
        all_metrics['mae'].append(mae)
        all_metrics['mse'].append(mse)
        all_metrics['rmse'].append(rmse)
        all_metrics['psnr'].append(psnr_score)
        all_metrics['ssim'].append(ssim_score)
        
        if idx < 20:
            fig, axes = plt.subplots(1, 4, figsize=(16, 4))
            
            cond_vis = (condition.cpu().squeeze().numpy() + 1) / 2
            axes[0].imshow(cond_vis, cmap='gray')
            axes[0].set_title('Pre-Op')
            axes[0].axis('off')
            
            axes[1].imshow(target_01.squeeze(), cmap='gray')
            axes[1].set_title('Ground Truth')
            axes[1].axis('off')
            
            axes[2].imshow(pred_01.squeeze(), cmap='gray')
            axes[2].set_title('Generated')
            axes[2].axis('off')
            
            diff = np.abs(target_01.squeeze() - pred_01.squeeze())
            axes[3].imshow(diff, cmap='hot')
            axes[3].set_title('Difference')
            axes[3].axis('off')
            
            fig.suptitle(f"Sample {idx+1} | MAE: {mae:.4f} | SSIM: {ssim_score:.4f} | PSNR: {psnr_score:.2f} dB")
            plt.tight_layout()
            plt.savefig(os.path.join(config.TEST_RESULTS_DIR, f"test_{idx+1:03d}.png"), dpi=150)
            plt.close()
    
    print("\n" + "="*70)
    print("FINAL TEST RESULTS")
    print("="*70)
    print(f"MAE:  {np.mean(all_metrics['mae']):.6f} ± {np.std(all_metrics['mae']):.6f}")
    print(f"MSE:  {np.mean(all_metrics['mse']):.6f} ± {np.std(all_metrics['mse']):.6f}")
    print(f"RMSE: {np.mean(all_metrics['rmse']):.6f} ± {np.std(all_metrics['rmse']):.6f}")
    print(f"SSIM: {np.mean(all_metrics['ssim']):.4f} ± {np.std(all_metrics['ssim']):.4f}")
    print(f"PSNR: {np.mean(all_metrics['psnr']):.2f} ± {np.std(all_metrics['psnr']):.2f} dB")
    print("="*70)

# ===== CHECKPOINT MANAGEMENT =====
def save_checkpoint(epoch, model, optimizer, scheduler, scaler, losses, val_metrics_history, ema=None):
    checkpoint_path = os.path.join(config.MODELS_DIR, 'latest_checkpoint.pth')
    checkpoint = {
        'epoch': epoch,
        'model_state_dict': model.state_dict(),
        'optimizer_state_dict': optimizer.state_dict(),
        'scheduler_state_dict': scheduler.state_dict(),
        'scaler_state_dict': scaler.state_dict(),
        'losses': losses,
        'val_metrics_history': val_metrics_history
    }
    
    if ema is not None:
        checkpoint['ema_shadow'] = ema.shadow
    
    torch.save(checkpoint, checkpoint_path)
    
    if epoch % 50 == 0 or epoch == config.NUM_EPOCHS:
        epoch_checkpoint_path = os.path.join(config.MODELS_DIR, f'checkpoint_epoch_{epoch}.pth')
        torch.save(checkpoint, epoch_checkpoint_path)
    
    print(f"✓ Checkpoint saved at epoch {epoch}")

def load_checkpoint(model, optimizer, scheduler, scaler, ema=None):
    checkpoint_path = os.path.join(config.MODELS_DIR, 'latest_checkpoint.pth')
    
    if not os.path.exists(checkpoint_path):
        print("No checkpoint found. Starting from scratch.")
        return 0, [], []
    
    print(f"Loading checkpoint from {checkpoint_path}...")
    checkpoint = torch.load(checkpoint_path, map_location=config.DEVICE, weights_only=False)

    # ------------------------
    # MODEL LOAD (SAFE)
    # ------------------------
    missing, unexpected = model.load_state_dict(
        checkpoint['model_state_dict'], 
        strict=False
    )

    if unexpected:
        print("\n⚠️ Ignored unexpected keys:")
        for k in unexpected:
            print("   -", k)

    if missing:
        print("\n⚠️ Missing keys:")
        for k in missing:
            print("   -", k)

    # ------------------------
    # OPTIMIZER LOAD (TRY / EXCEPT)
    # ------------------------
    try:
        optimizer.load_state_dict(checkpoint['optimizer_state_dict'])
    except ValueError:
        print("⚠️ Optimizer groups changed — skipping optimizer load.")

    # ------------------------
    # SCHEDULER & SCALER LOAD
    # ------------------------
    try:
        scheduler.load_state_dict(checkpoint['scheduler_state_dict'])
    except Exception:
        print("⚠️ Scheduler state mismatch — skipping scheduler load.")

    try:
        scaler.load_state_dict(checkpoint['scaler_state_dict'])
    except Exception:
        print("⚠️ Scaler mismatch — starting fresh.")
    
    # ------------------------
    # EMA LOAD
    # ------------------------
    if ema is not None and 'ema_shadow' in checkpoint:
        ema.shadow = checkpoint['ema_shadow']
        print("✓ EMA weights loaded")

    print(f"\n✓ Resumed from epoch {checkpoint['epoch']}\n")
    return checkpoint['epoch'], checkpoint.get('losses', []), checkpoint.get('val_metrics_history', [])
